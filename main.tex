\documentclass{dudissertation}

% Ensure all of these values match your situation!
\title{Title}
\author{First Last}
\faculty{\dursecsfaculty} % Use \dursecsfaculty or \dunsmfaculty
\advisor{Advisor's Name}
\degreetype{Doctor of Philosophy}
\dissertationtype{Dissertation} % or {Thesis}
\graduationmonth{Month}
\graduationyear{Year}

% Copyright pages are optional. Comment to disable
\toggletrue{copyrightpage}

\abstractfilename{abstract}

% Acknowledgments pages are optional. Comment to disable
\acknowledgments{acknowledgments.tex}

% Algorithms list. Comment out to disable
\toggletrue{listofalgorithms}

% List of Figures. As per DU rules, required for 5+, optional otherwise
\toggletrue{listoffigures}

% List of Tables. As per DU rules, required for 5+, optional otherwise
\toggletrue{listoftables}

% Setup the bibliography
\usepackage[
backend=biber,
style=numeric
]{biblatex}

\addbibresource{bibliography.bib}

% Create an index. To disable, comment out below line and designated block
% near end of this document
\enableindex

% Setup asmthm environments
\theoremstyle{plain}
\newtheorem{thm}{Theorem}[section]
\newtheorem{lem}[thm]{Lemma}
\newtheorem{prop}[thm]{Proposition}
\newtheorem{cor}[thm]{Corollary}

\theoremstyle{definition}
\newtheorem{defn}[thm]{Definition}

\newtheorem{exmp}[thm]{Example}

\theoremstyle{remark}
\newtheorem{rem}[thm]{Remark}
\newtheorem{note}[thm]{Note}

% cleveref setup
% Most environments (including the amsthm setup above) should just work.
\crefname{algorithm}{algorithm}{algorithms}

% Put any additional package declarations here

% While writing, it is recommended to enable this to highlight overfull boxes
% and put line numbers in the margins. Comment out for final version
% Warning: Enabling/disabling draft mode may require deleting intermediary
% files (main.aux, etc). This can be done with the command latexmk -C
\enabledraftmode

% If you wish to see lines at the exact margins, uncomment the line below
\geometry{showframe}

% List of Symbols. Comment out to disable.
% Note if you define custom command for symbols, ensure they occur prior
% to this line
\setuplistofsymbols{symbols.tex}

\begin{document}
% Create all the various required pages (title, copyright, abstract, etc)
% This line is required
\startofdocument

%Include all your chapters
\include{ch_foo}
\include{ch_bar}
\include{ch_algo}
\include{ch_ref}

% Put the bibliography into the table of contents. This really shouldn't be
% needed, the \KOMAoption{bibliography{totoc} should take care of this, but
% for some reason biblatex is not recognizing that option and emitting the
% line below as appropriate. It is not clear if the problem is how the
% KOMA-Script option is being specified, a problem in KOMA-Script, or in
% biblatex.
\addcontentsline{toc}{chapter}{Bibliography}
\printbibliography

% Print out the index. If you do not want an index comment out the following 2
% lines
\clearpage
\printindex

\end{document}
